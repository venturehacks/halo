  <div className="mb-4">

    />
    <Tag

        alert('tag clicked!');
      }}
      label="Clickable tag!"
    />
    <Tag color="blue" onClose={() => {alert("close clicked!")}} >

      <span className="text-red-700">Second Element</span>

  </div>

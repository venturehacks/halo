# Tag

Successor to `PillTag` `Flair` (deprecated).

### Usage

<Playground>
  <div className="mb-4">
    <InterfaceHeader title="Non-Interactive" section flow />
    <Tag color="red" label="Red tag" />
    <Tag color="gray" label="Gray tag underline" labelDecoration="underline" />
    <Tag
      color="blue"
      icon={<BookmarkOutlineIcon />}
      label="Blue tag with icon"
    />
    <Tag color="orange" icon={<BookmarkOutlineIcon />} size="sm" />
  </div>

  <div className="mb-4">
    <InterfaceHeader title="Interactive" section flow />
    <Tag onClose={() => {alert("close clicked!")}} >
      <span className="font-medium">Tag Name</span>
      <span className="text-green-400">Flair</span>
    </Tag>
    <Tag
      onClose={() => {alert("close clicked!")}}
      onClick={() => {alert("tag clicked!")}}
      color="gray"
    >
      <span className="text-dark-aaa font-medium">Clickable Tag</span>
      <span className="text-green-400">Flair</span>
    </Tag>
    <Tag
      color="green"
      onClose={() => {
        alert('close clicked!');
      }}
      label="Green tag with clickable close icon"
    />
    <Tag
      color="purple"
      onClick={() => {
        alert('tag clicked!');
      }}
      label="Clickable tag!"
    />
  </div>
</Playground>

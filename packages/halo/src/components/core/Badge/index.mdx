# Badge

Single character colored badge to decorate an element.

### Usage

<Playground>
  <Box row>
    <Box relative margin={{ right: 2 }}>
      <PillTag>square, sm, top</PillTag>
      <Badge color="purple" shape="square">ABC</Badge>
    </Box>
    <Box relative margin={{ right: 2 }}>
      <PillTag>square, sm, bottom</PillTag>
      <Badge color="purple" position="bottom" shape="square">DEF</Badge>
    </Box>
  </Box>
  <Box row padding={{ top: 2 }}>
    <Box relative margin={{ right: 2 }}>
      <PillTag>square, md, top</PillTag>
      <Badge color="blue" shape="square" size="md">RST</Badge>
    </Box>
    <Box relative margin={{ right: 2 }}>
      <PillTag>square, md, bottom</PillTag>
      <Badge color="blue" position="bottom" shape="square" size="md">XYZ</Badge>
    </Box>
  </Box>
  <Box row padding={{ top: 2 }}>
    <Box relative margin={{ right: 2 }}>
      <PillTag>circle, sm, top</PillTag>
      <Badge color="red" size="sm">A</Badge>
    </Box>
    <Box relative margin={{ right: 2 }}>
      <PillTag>circle, sm, bottom</PillTag>
      <Badge color="red" position="bottom">B</Badge>
    </Box>
  </Box>
  <Box row padding={{ top: 2 }}>
    <Box relative margin={{ right: 2 }}>
      <PillTag>circle, md, top</PillTag>

  </Box>
</Playground>

### Exports

```ts
type BadgeColor =  'blue' | 'gray' | 'green' | 'orange' | 'purple' | 'red';

type BadgeShape = 'square' | 'circle';
```

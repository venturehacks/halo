# Select

Raw input fields are designed to be controlled by a [smart, form-connected parent](https://github.com/venturehacks/AngelList/blob/master/frontend/talent/components/Form/TextField/index.tsx).

### Usage

<Playground>
  <Label
    field="title"
    title="Job Title"
    byline="Select your position"
    containsFieldGroup
  >
    <div className="max-w-lg space-x-8">
      <Select
        name="title"
        value="software engineer"
        options={[
          { label: 'Software Engineer', value: 'eng' },
          { label: 'Product Manager', value: 'pm]' },
          { label: 'Head of Sales', value: 'hs' },
        ]}
        onChange={() => {}}
      />
      <Select
        name="title"
        value="oops, error"
        options={[
          { label: 'Product Manager', value: 'pm]' },
          { label: 'Software Engineer', value: 'eng' },
          { label: 'Head of Sales', value: 'hs' },
        ]}
        onChange={() => {}}
        hasError
      />
    </div>
  </Label>
</Playground>

### Options

```jsx
import { Select, SelectOption } from 'halo';

const options: SelectOption[] = [
  { label: 'Chocolate', value: 'chocolate' },
  { label: 'Vanilla', value: 'vanilla' },
  { label: 'Strawberry', value: 'strawberry' },
];
```

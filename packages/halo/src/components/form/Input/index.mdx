# Input

Raw input fields are designed to be controlled by a [smart, form-connected parent](https://github.com/venturehacks/AngelList/blob/master/frontend/talent/components/Form/TextField/index.tsx).

### Usage

<Playground>
  <div className="max-w-xl">
    <Label field="foo" title="Text Input">
      <Input type="text" name="foo" placeholder="Your name" maxLength={100} />
    </Label>
    <Bracket>
      <Label field="foo" title="Icon Left">
        <Input
          type="text"
          name="foo"

          maxLength={100}
          icon={<TimeIcon />}
        />
      </Label>
      <Label field="foo" title="Icon Right">
        <Input
          type="text"
          name="foo"
          placeholder="archive@foo.com"
          maxLength={100}
          iconPosition="right"
          icon={<ArchiveIcon />}
        />
      </Label>
    </Bracket>
    <Bracket>
      <Label field="bar" title="Email">
        <Input
          type="email"
          name="bar"
          onChange={() => {}}
          value="What is email?"
          intrinsicWidth="auto"
          hasError
        />
        <FieldErrorMessage
          errorSeverity="warning"
          message="An email is required"
        />
      </Label>
      <Label field="password" title="Password">
        <Input
          type="password"
          name="password"
          intrinsicWidth="auto"
          value="abcdefghijk"
          onChange={() => {}}

          hasError
        />
        <FieldErrorMessage errorSeverity="critical" message="Does not match" />
      </Label>
    </Bracket>
    <Label field="phone" title="Number">
      <Input
        type="number"
        name="phone"
        intrinsicWidth="auto"

        onChange={() => {}}
      />
    </Label>
  </div>
